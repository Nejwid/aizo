

using namespace std;

// Funkcja pomocnicza do naprawy kopca
template <typename T>
void HeapSort<T>::Fix(T arr[], int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    // Sprawdzanie lewego dziecka
    if (left < n && arr[left] > arr[largest])
        largest = left;

    // Sprawdzanie prawego dziecka
    if (right < n && arr[right] > arr[largest])
        largest = right;

    // Zamiana i dalsze naprawianie, jeli potrzebne
    if (largest != i) {
        swap(arr[i], arr[largest]);
        Fix(arr, n, largest);
    }
}

template <typename T>
HeapSort<T>::HeapSort(T arr[], int s) {
    this->array = arr;
    this->size = s;
}

template <typename T>
bool HeapSort<T>::Sort(T arr[], int n) {
    // Budowanie kopca
    for (int i = n / 2 - 1; i >= 0; i--)
        Fix(arr, n, i);

    // Wyci¿ganie elementµw z kopca
    for (int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]);
        Fix(arr, i, 0);
    }
    return true;
}
