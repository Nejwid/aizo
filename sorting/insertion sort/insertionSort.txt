
using namespace std;

template <typename T>
InsertionSort<T>::InsertionSort(T arr[], int s) {
    this->array = arr;
    this->size = s;
}

template <typename T>
bool InsertionSort<T>::Sort(T arr[], int n) {
    for (int i = 1; i < n; i++) {
        T key = arr[i]; // Klucz uogólnionego typu
        int j = i - 1;

        // Przesuwanie elementów wiêkszych od klucza w prawa strone - zagniezdzona petla o z³o¿onosci O(n*n)
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            --j;
        }
        arr[j + 1] = key;
    }
    return true;
}
